常规配置参数使用演示

  1. 普通调用，不带参数

    var a = $.dialog();

  2. 设置ID值，防止重复弹出，更便于日后取得窗口对象

    var a = $.dialog({ id: "a1", lock:true });

  3. 支持HTML内容,支持page外部页面(*)

    /** content: html内容 */
var a = $.dialog({ 
    id: "a2" ,
    content: "我是HTML内容"
});
    (增强拖动体验只能全局配置：extendDrag: true) 
    /** content: url:外部页面 */
var a = $.dialog({ 
    id: "a3_1" ,
    title: "欢迎使用手机帐族",
    content: "url:http://m.zonezu.com/",
    width: 200,
    height: 200,
    minWidth: 320,
    minHeight: 450
});

  4. 通过title属性更改标题内容

    var a = $.dialog({ 
    id: "a3" ,
    content: "我是HTML内容",
    title: "我是新的Title"
});

  5. 可以自由设置titleIcon参数，改变标题栏左上角的图标

    var a = $.dialog({ 
    id: "a_icon" ,
    titleIcon: "",
    content: "我可以自由定制标题栏小图标哦",
    title: "看我左边的图标"
});

  6. 添加自定义按钮

    var a = $.dialog({
	var timer = 3;
    id: "a5" ,
    content: "我是HTML内容",
    button: [
    	{
            name:"上一步",
            callback: function () {
                this.content("这是上一步的内容");
                return false;
            },
            focus: true
        },
        {
            name:"下一步",
            callback: function () {
                this.content('你点击了下一步,3秒后关闭');
                this.time(timer, function(){ 
                    $.dialog.list["a5"].title("窗口将在 " + timer-- + " 秒后关闭");
                });
                return false;
            }
        },
        {
            name: "无效按钮",
            disabled: true
        },
    ]
});

  7. 确定按钮回调函数，可通过okVal改变ok按钮的文本

    var a = $.dialog({ 
    id: "a5" ,
    content: "我是HTML内容",
    ok: function(){
    	alert("你点击了确定按钮");
        this.content("通过返回值true,执行默认函数，返回false，可以阻止窗口关闭");
		return false;
    },
    okVal:"我是OK按钮"
});

  8. 取消按钮回调函数,可通过cancelVal改变cancel按钮的文本

    var a = $.dialog({
    id: "a7" ,
    content: "我是HTML内容,我只有一个取消按钮，右上角关闭按钮，和右下角取消按钮，同属一个函数控制。",	
    cancel:function(){
        alert("你点击了取消按钮");
    },
    cancelVal:"我是取消按钮"
});

  9. 对话框初始化后执行的函数 init

    var a = $.dialog({
    id: "a8" ,
    content:"init对话框初始化后执行的函数",
    init:function(){
        alert("我是对话框初始化执行的函数");
    }
});

  10. 对话框关闭前执行的函数

    var a = $.dialog({
    id: "a9",
    content: "对话框关闭前可执行回调函数close",

    	alert("等我执行完了才关闭，可用户关闭后刷新父页面");
    }	
});
